#include <SoftwareSerial.h>

SoftwareSerial ESPserial(5, 6); // RX | TX
String address = "http://192.168.0.181:3000/agents/1";
void Get(String address)
{
    ESPserial.println("GET," + address);
    while (1)
    {
        if (ESPserial.available())
        {
            String payload = ESPserial.readStringUntil('\n');
            Serial.println(payload);
            break;
        }
        else
        {
          Serial.println("NO");
        }
    }
}
void setup()
{
    Serial.begin(115200);
    ESPserial.begin(4800);
}

void loop()
{
    Get(address);
}
